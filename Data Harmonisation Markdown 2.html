plot_missing_count <- function(varname, rf=rf0){
  data <- rf[,colnames(rf)==varname, drop = TRUE]
  data[data!=888] <- 0
  freq(factor(data, levels = c(0,888), labels = c("Non-Missing", "Missing")), cumul = F)
}

plot_preg <- function(varname, rf=rf0){
  count <- str_sub(varname,-1)
  data <- rf[,colnames(rf)==varname, drop = TRUE]
  data[!data%in%c(666,777,888)] <- 0
  freq(factor(data, levels = c(0,666,777,888), labels = c("Non-Missing", paste0("<", count, " Children"), "Nonparous", "Missing")), cumul = F)
}

plot_usage <- function(varname, rf=rf0){
  type <- str_sub(varname, -4)
  data <- rf[,colnames(rf)==varname, drop = TRUE]
  data[!data%in%c(666,777,888)] <- 0
  if(type == "stop"){
    freq(factor(data, levels = c(0,666,777,888), labels = c("Non-Missing", "Current", "Never Started", "Missing")), cumul = F)
  } else {
    freq(factor(data, levels = c(0,777,888), labels = c("Non-Missing", "Never Started", "Missing")), cumul = F)
  }
}
